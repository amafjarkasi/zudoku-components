<script>{`
  document.addEventListener('DOMContentLoaded', function() {
    const yesBtn = document.querySelector('.feedback-btn--yes');
    const noBtn = document.querySelector('.feedback-btn--no');
    const responseYes = document.querySelector('.feedback-response--yes');
    const responseNo = document.querySelector('.feedback-response--no');
    const submitBtn = document.querySelector('.feedback-submit');

    yesBtn.addEventListener('click', function() {
      yesBtn.classList.add('selected');
      noBtn.classList.remove('selected');
      responseYes.style.display = 'block';
      responseNo.style.display = 'none';
    });

    noBtn.addEventListener('click', function() {
      noBtn.classList.add('selected');
      yesBtn.classList.remove('selected');
      responseNo.style.display = 'block';
      responseYes.style.display = 'none';
    });

    if (submitBtn) {
      submitBtn.addEventListener('click', function() {
        const textarea = document.querySelector('.feedback-textarea');
        submitBtn.textContent = 'Sent!';
        submitBtn.disabled = true;
        textarea.disabled = true;
      });
    }
  });
`}</script>
