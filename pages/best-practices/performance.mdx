## Caching Strategies

### Cache Frequently Accessed Data

```typescript
import { MemoryCache } from "@premium-api/cache";

const cache = new MemoryCache({ ttl: 300 }); // 5 minute TTL

async function getProductWithCache(productId: string) {
  const cacheKey = `product:${productId}`;

  // Check cache first
  const cached = cache.get(cacheKey);
  if (cached) return cached;

  // Fetch from API
  const product = await client.products.get(productId);

  // Store in cache
  cache.set(cacheKey, product);

  return product;
}
```

### Cache Product Catalogs

```python
response = client.products.list(
    limit=100,
    cache=True,
    cache_ttl=300
)

import redis

r = redis.Redis(host='localhost', port=6379)

def get_products_cached():
    cached = r.get('products:list')
    if cached:
        return json.loads(cached)

    products = client.products.list()
    r.setex('products:list', 300, json.dumps(products))
    return products
```

## Request Optimization

### Batch Requests

```typescript
// Instead of multiple requests
const [products, customers, orders] = await Promise.all([
  client.products.list({ limit: 100 }),
  client.customers.list({ limit: 100 }),
  client.orders.list({ limit: 100 }),
]);

// Use batch endpoint when available
const batch = await client.batch.execute([
  { operation: "products.list", params: { limit: 100 } },
  { operation: "customers.list", params: { limit: 100 } },
  { operation: "orders.list", params: { limit: 100 } },
]);
```

### Use Field Filtering

```typescript
// Only fetch needed fields
const orders = await client.orders.list({
  fields: ["id", "status", "total", "created_at"],
  limit: 50,
});

// Reduces response size by ~70%
```

### Pagination Strategy

```python
cursor = None
all_products = []

while True:
    page = client.products.list(
        limit=250,
        cursor=cursor
    )

    all_products.extend(page['items'])

    cursor = page['next_cursor']
    if not cursor:
        break

last_sync = get_last_sync_time()
new_products = client.products.list(
    updated_after=last_sync,
    limit=250
)
```

```python
def handle_order_created(order):
```

## Performance Monitoring

### Track API Latency

```typescript
const perf = require("perf_hooks");

function measureApiCall<T>(fn: () => Promise<T>, label: string): Promise<T> {
  const start = perf.performance.now();

  return fn().then((result) => {
    const duration = perf.performance.now() - start;
    console.log(`${label}: ${duration.toFixed(2)}ms`);

    metrics.histogram("api.latency", duration, {
      operation: label,
    });

    return result;
  });
}

// Usage
const products = await measureApiCall(
  () => client.products.list(),
  "products.list",
);
```
