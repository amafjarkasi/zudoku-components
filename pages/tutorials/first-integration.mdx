## Prerequisites

- Node.js 18+ or Python 3.9+
- An API key with `products:read`, `orders:write`, and `analytics:write` permissions
- Your favorite HTTP client (fetch, axios, or requests)

```javascript
  async request(method, endpoint, body = null) {
  const url = `${this.baseUrl}${endpoint}`;
  const headers = {
  Authorization: `Bearer ${this.apiKey}`,
  "Content-Type": "application/json",
  };

  method,
  headers,
  };

      }
````

```python
# src/client.py
import os
import requests
from dotenv import load_dotenv

load_dotenv()

API_KEY = os.getenv('ZEPHYR_API_KEY')
API_URL = os.getenv('ZEPHYR_API_URL', 'https://api.zephyr.com/v1')


class ZephyrClient:
    """Zephyr API Client - Handles authentication and requests"""

    def __init__(self, api_key, base_url=API_URL):
        self.api_key = api_key
        self.base_url = base_url

    def _request(self, method, endpoint, body=None):
        """Make an authenticated API request"""
        url = f"{self.base_url}{endpoint}"
        headers = {
            'Authorization': f'Bearer {self.api_key}',
            'Content-Type': 'application/json'
        }

        try:
            response = requests.request(
                method=method,
                url=url,
                headers=headers,
                json=body
            )
            response.raise_for_status()
            return response.json()
        except requests.exceptions.RequestException as e:
            print(f"Request failed: {method} {endpoint}")
            raise

    def get(self, endpoint):
        return self._request('GET', endpoint)

    def post(self, endpoint, body):
        return self._request('POST', endpoint, body)

    def put(self, endpoint, body):
        return self._request('PUT', endpoint, body)

    def delete(self, endpoint):
        return self._request('DELETE', endpoint)


# Export singleton instance
client = ZephyrClient(API_KEY)
````

## Step 3: Fetch Products

Now let's fetch the available products from your catalog:

### Node.js

```javascript
// src/index.js
import { client } from "./client.js";

/**
 * Fetch active products with pagination
 * @param {number} page - Page number
 * @param {number} perPage - Items per page
 */
async function getProducts(page = 1, perPage = 20) {
  console.log(`Fetching products (page ${page})...`);

  const response = await client.get(
    `/products?page=${page}&per_page=${perPage}&status=active`,
  );

  console.log(`Found ${response.meta.total} products total`);
  return response.data;
}

// Run the function
const products = await getProducts();
console.log(`First 5 products:`, products.slice(0, 5));
```

With a product selected, let's create an order:

### Node.js

```javascript
// src/index.js (continued)

/**
 * Create a new order
 * @param {string} productId - Product ID
 * @param {string} variantId - Variant ID
 * @param {number} quantity - Quantity to order
 * @param {object} customer - Customer information
 */
async function createOrder(productId, variantId, quantity, customer) {
  console.log(`Creating order for customer ${customer.email}...`);

  const orderData = {
    line_items: [
      {
        variant_id: variantId,
        quantity: quantity,
        price: 199.99, // Price at time of order

      last_name: customer.lastName,
      address: {
        line1: customer.address,
        city: customer.city,
        state: customer.state,
```

## Step 5: Track Analytics Event

Now let's track the purchase event for analytics:

### Node.js

```javascript
// src/index.js (continued)

/**
 * Track a purchase event in analytics
 * @param {object} order - The order object
 * @param {object} customer - Customer information
 */
async function trackPurchaseEvent(order, customer) {
  console.log(`Tracking purchase event for order ${order.id}...`);

  const eventData = {
    event_name: "purchase_completed",
    user_id: customer.email, // Using email as user identifier
    properties: {
      order_id: order.id,
      revenue: order.pricing.grand_total,
      tax: order.pricing.tax,
      shipping: order.pricing.shipping,
      discount: order.pricing.discount,
      payment_method: order.payment.method,
      items: order.line_items.map((item) => ({
        product_id: item.variant_id,
        name: item.product_name,
        variant_name: item.variant_name,
        quantity: item.quantity,
        unit_price: item.unit_price,
      })),
    },
    context: {
      ip: "203.0.113.42", // Customer's IP address
      user_agent: "Mozilla/5.0 Tutorial/1.0",
      locale: "en-US",
      page_url: "https://example.com/checkout/complete",
    },
  };

  const response = await client.post("/analytics/events", eventData);
  console.log(`Event tracked: ${response.event_id}`);

  return response;
}

// Track the event after creating the order
const analyticsResult = await trackPurchaseEvent(order, customer);
console.log("Analytics event ID:", analyticsResult.event_id);
```

## Step 6: Complete Tutorial Script

Here's the complete Node.js implementation:

```javascript
// src/index.js - Complete tutorial integration
import { client } from "./client.js";

async function main() {
  console.log("=== Zephyr API Tutorial Integration ===\n");

  try {
    // Step 1: Fetch products
    console.log("Step 1: Fetching products...");
    const productsResponse = await client.get(
      "/products?page=1&per_page=20&status=active",
    );
    const products = productsResponse.data;
    console.log(`Found ${productsResponse.meta.total} active products\n`);

    if (products.length === 0) {
      throw new Error("No products found. Please create products first.");
    }

    // Step 2: Select a product and variant
    const product = products[0];
    const variant = product.variants[0];
    console.log(`Selected: ${product.name} (${variant.name})`);
    console.log(`Price: $${variant.price}\n`);

    // Step 3: Create customer
    const customer = {
      email: "sarah.chen@example.com",
      firstName: "Sarah",
      lastName: "Chen",
      address: "123 Main Street",
      city: "San Francisco",
      state: "CA",
      postalCode: "94102",
      country: "US",
    };

    // Step 4: Create order
    console.log("Step 2: Creating order...");
    const orderData = {
      line_items: [
        {
          variant_id: variant.id,
          quantity: 1,
          price: variant.price,
        },
      ],
      customer: {
        email: customer.email,
        first_name: customer.firstName,
        last_name: customer.lastName,
        address: {
          line1: customer.address,
          city: customer.city,
          state: customer.state,
          postal_code: customer.postalCode,
          country: customer.country,
        },
      },
      shipping: {
        method: "standard",
        price: 9.99,
      },
      payment_method: "card",
      metadata: {
        source: "tutorial_integration",
      },
    };

    const order = await client.post("/orders", orderData);
    console.log(`Order created: ${order.id}`);
    console.log(`Total: $${order.pricing.grand_total}\n`);

    // Step 5: Track analytics event
    console.log("Step 3: Tracking analytics event...");
    const eventData = {
      event_name: "purchase_completed",
      user_id: customer.email,
      properties: {
        order_id: order.id,
        revenue: order.pricing.grand_total,
        items: order.line_items.length,
      },
    };

    const event = await client.post("/analytics/events", eventData);
    console.log(`Event tracked: ${event.event_id}\n`);

    // Success!
    console.log("=== Integration Complete! ===");
    console.log(`Order: ${order.id}`);
    console.log(`Analytics: ${event.event_id}`);
  } catch (error) {
    console.error("Integration failed:", error.message);
    process.exit(1);
  }
}

main();
```

## Error Handling

Add robust error handling to your integration:

```javascript
// src/error-handler.js

class ZephyrAPIError extends Error {
  constructor(errorResponse, statusCode) {
    super(errorResponse.error?.message || "Unknown error");
    this.statusCode = statusCode;
    this.code = errorResponse.error?.code;
    this.details = errorResponse.error?.details;
    this.requestId = errorResponse.error?.request_id;
  }
}

async function safeApiCall(apiFunction, ...args) {
  try {
    return await apiFunction(...args);
  } catch (error) {
    if (error instanceof ZephyrAPIError) {
      // Handle known API errors
      switch (error.code) {
        case "rate_limit_exceeded":
          console.error("Rate limit hit - implement backoff");
          // Implement exponential backoff
          await sleep(1000);
          return safeApiCall(apiFunction, ...args);

        case "invalid_api_key":
          console.error("Invalid API key - check your configuration");
          throw error;

        case "insufficient_permissions":
          console.error("Missing permissions - update API key scopes");
          throw error;

        default:
          console.error(`API Error [${error.code}]: ${error.message}`);
          throw error;
      }
    }
    // Network or other errors
    console.error("Network error:", error.message);
    throw error;
  }
}

function sleep(ms) {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

// Usage
const products = await safeApiCall(() => client.get("/products?status=active"));
```
